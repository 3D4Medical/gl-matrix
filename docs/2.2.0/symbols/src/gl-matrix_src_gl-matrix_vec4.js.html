 29 /**
 30  * Creates a new, empty vec4
 31  *
 32  * @returns {vec4} a new 4D vector
 33  */
 34 vec4.create = function() {
 35     var out = new GLMAT_ARRAY_TYPE(4);
 36     out[0] = 0;
 37     out[1] = 0;
 38     out[2] = 0;
 39     out[3] = 0;
 40     return out;
 41 };
 42 
 43 /**
 44  * Creates a new vec4 initialized with values from an existing vector
 45  *
 46  * @param {vec4} a vector to clone
 47  * @returns {vec4} a new 4D vector
 48  */
 49 vec4.clone = function(a) {
 50     var out = new GLMAT_ARRAY_TYPE(4);
 51     out[0] = a[0];
 52     out[1] = a[1];
 53     out[2] = a[2];
 54     out[3] = a[3];
 55     return out;
 56 };

 59  * Creates a new vec4 initialized with the given values

256 /**
257  * Calculates the euclidian distance between two vec4's
258  *
259  * @param {vec4} a the first operand
260  * @param {vec4} b the second operand
261  * @returns {Number} distance between a and b
262  */
263 vec4.distance = function(a, b) {
264     var x = b[0] - a[0],
265         y = b[1] - a[1],
266         z = b[2] - a[2],
267         w = b[3] - a[3];
268     return Math.sqrt(x*x + y*y + z*z + w*w);
269 };

353 /**
354  * Normalize a vec4
355  *
356  * @param {vec4} out the receiving vector
357  * @param {vec4} a vector to normalize
358  * @returns {vec4} out
359  */
360 vec4.normalize = function(out, a) {
361     var x = a[0],
362         y = a[1],
363         z = a[2],
364         w = a[3];
365     var len = x*x + y*y + z*z + w*w;
366     if (len > 0) {
367         len = 1 / Math.sqrt(len);
368         out[0] = a[0] * len;
369         out[1] = a[1] * len;
370         out[2] = a[2] * len;
371         out[3] = a[3] * len;
372     }
373     return out;
374 };

387 /**
388  * Performs a linear interpolation between two vec4's
389  *
390  * @param {vec4} out the receiving vector
391  * @param {vec4} a the first operand
392  * @param {vec4} b the second operand
393  * @param {Number} t interpolation amount between the two inputs
394  * @returns {vec4} out
395  */
396 vec4.lerp = function (out, a, b, t) {
397     var ax = a[0],
398         ay = a[1],
399         az = a[2],
400         aw = a[3];
401     out[0] = ax + t * (b[0] - ax);
402     out[1] = ay + t * (b[1] - ay);
403     out[2] = az + t * (b[2] - az);
404     out[3] = aw + t * (b[3] - aw);
405     return out;
406 };

428 /**
429  * Transforms the vec4 with a mat4.
430  *
431  * @param {vec4} out the receiving vector
432  * @param {vec4} a the vector to transform
433  * @param {mat4} m matrix to transform with
434  * @returns {vec4} out
435  */
436 vec4.transformMat4 = function(out, a, m) {
437     var x = a[0], y = a[1], z = a[2], w = a[3];
438     out[0] = m[0] * x + m[4] * y + m[8] * z + m[12] * w;
439     out[1] = m[1] * x + m[5] * y + m[9] * z + m[13] * w;
440     out[2] = m[2] * x + m[6] * y + m[10] * z + m[14] * w;
441     out[3] = m[3] * x + m[7] * y + m[11] * z + m[15] * w;
442     return out;
443 };

445 /**
446  * Transforms the vec4 with a quat
447  *
448  * @param {vec4} out the receiving vector
449  * @param {vec4} a the vector to transform
450  * @param {quat} q quaternion to transform with
451  * @returns {vec4} out
452  */
453 vec4.transformQuat = function(out, a, q) {
454     var x = a[0], y = a[1], z = a[2],
455         qx = q[0], qy = q[1], qz = q[2], qw = q[3],
456 
457         // calculate quat * vec
458         ix = qw * x + qy * z - qz * y,
459         iy = qw * y + qz * x - qx * z,
460         iz = qw * z + qx * y - qy * x,
461         iw = -qx * x - qy * y - qz * z;
462 
463     // calculate result * inverse quat
464     out[0] = ix * qw + iw * -qx + iy * -qz - iz * -qy;
465     out[1] = iy * qw + iw * -qy + iz * -qx - ix * -qz;
466     out[2] = iz * qw + iw * -qz + ix * -qy - iy * -qx;
467     return out;
468 };

470 /**
471  * Perform some operation over an array of vec4s.
472  *
473  * @param {Array} a the array of vectors to iterate over
474  * @param {Number} stride Number of elements between the start of each vec4. If 0 assumes tightly packed
475  * @param {Number} offset Number of elements to skip at the beginning of the array
476  * @param {Number} count Number of vec2s to iterate over. If 0 iterates over entire array
477  * @param {Function} fn Function to call for each vector in the array
478  * @param {Object} [arg] additional argument to pass to fn
479  * @returns {Array} a
480  * @function
481  */
482 vec4.forEach = (function() {
483     var vec = vec4.create();
484 
485     return function(a, stride, offset, count, fn, arg) {
486         var i, l;
487         if(!stride) {
488             stride = 4;
489         }
490 
491         if(!offset) {
492             offset = 0;
493         }
494         
495         if(count) {
496             l = Math.min((count * stride) + offset, a.length);
497         } else {
498             l = a.length;
499         }
500 
501         for(i = offset; i < l; i += stride) {
502             vec[0] = a[i]; vec[1] = a[i+1]; vec[2] = a[i+2]; vec[3] = a[i+3];
503             fn(vec, vec, arg);
504             a[i] = vec[0]; a[i+1] = vec[1]; a[i+2] = vec[2]; a[i+3] = vec[3];
505         }
506         
507         return a;
508     };
509 })();
